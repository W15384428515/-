    <script>
        // 标签切换功能
        document.querySelectorAll('.tab').forEach(tab => {
            tab.addEventListener('click', () => {
                document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
                document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
                
                tab.classList.add('active');
                const tabId = tab.getAttribute('data-tab');
                document.getElementById(tabId).classList.add('active');
            });
        });
        
        // 状态切换功能
        document.querySelectorAll('.status').forEach(status => {
            status.addEventListener('click', () => {
                if (status.classList.contains('purchased')) {
                    status.classList.remove('purchased');
                    status.classList.add('not-purchased');
                    status.textContent = status.textContent.includes('购买') ? '未购买' : '未完成';
                } else {
                    status.classList.remove('not-purchased');
                    status.classList.add('purchased');
                    status.textContent = status.textContent.includes('购买') ? '已购买' : '已完成';
                }
                updateProgress();
            });
        });
        
        // 搜索功能
        document.getElementById('searchBtn').addEventListener('click', performSearch);
        document.getElementById('searchInput').addEventListener('keyup', (e) => {
            if (e.key === 'Enter') performSearch();
        });
        
        function performSearch() {
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            if (!searchTerm) {
                // 清除高亮
                document.querySelectorAll('table tbody tr').forEach(row => {
                    row.style.backgroundColor = '';
                });
                return;
            }
            
            // 在所有表格中搜索
            document.querySelectorAll('table tbody tr').forEach(row => {
                let found = false;
                row.querySelectorAll('td').forEach(cell => {
                    if (cell.textContent.toLowerCase().includes(searchTerm)) {
                        found = true;
                    }
                });
                
                if (found) {
                    row.style.backgroundColor = '#fff3cd';
                    // 滚动到第一个匹配项
                    if (!document.querySelector('.highlighted')) {
                        row.classList.add('highlighted');
                        row.scrollIntoView({ behavior: 'smooth', block: 'center' });
                        setTimeout(() => row.classList.remove('highlighted'), 2000);
                    }
                } else {
                    row.style.backgroundColor = '';
                }
            });
        }
        
        // 更新进度条
        function updateProgress() {
            const totalItems = document.querySelectorAll('.status').length;
            const completedItems = document.querySelectorAll('.status.purchased').length;
            const progressPercent = Math.round((completedItems / totalItems) * 100);
            
            document.getElementById('progressPercent').textContent = `${progressPercent}%`;
            document.getElementById('progressBar').style.width = `${progressPercent}%`;
        }
        
        // 复制表格数据功能
        document.getElementById('copyTableBtn').addEventListener('click', function() {
            let tableData = "同沙骑行烤肉活动物资清单\n\n";
            
            // 食材表格
            tableData += "=== 食材清单 ===\n";
            const ingredientRows = document.querySelectorAll('#ingredients table tbody tr');
            ingredientRows.forEach(row => {
                const cells = row.querySelectorAll('td');
                if (cells.length >= 2) {
                    tableData += `${cells[0].textContent} - ${cells[1].textContent} - ${cells[2].textContent}\n`;
                }
            });
            
            // 工具表格
            tableData += "\n=== 工具清单 ===\n";
            const toolRows = document.querySelectorAll('#tools table tbody tr');
            toolRows.forEach(row => {
                const cells = row.querySelectorAll('td');
                if (cells.length >= 4) {
                    tableData += `${cells[0].textContent} - ${cells[1].textContent} - ${cells[3].textContent}\n`;
                }
            });
            
            // 服务表格
            tableData += "\n=== 服务任务 ===\n";
            const serviceRows = document.querySelectorAll('#services table tbody tr');
            serviceRows.forEach(row => {
                const cells = row.querySelectorAll('td');
                if (cells.length >= 3) {
                    tableData += `${cells[0].textContent} - ${cells[2].textContent}\n`;
                }
            });
            
            // 复制到剪贴板
            if (navigator.clipboard && window.isSecureContext) {
                navigator.clipboard.writeText(tableData).then(() => {
                    showCopyNotice();
                });
            } else {
                // 备用方法
                const textArea = document.createElement("textarea");
                textArea.value = tableData;
                document.body.appendChild(textArea);
                textArea.select();
                document.execCommand('copy');
                document.body.removeChild(textArea);
                showCopyNotice();
            }
        });
        
        function showCopyNotice() {
            const notice = document.getElementById('copyNotice');
            notice.style.display = 'block';
            setTimeout(() => {
                notice.style.display = 'none';
            }, 3000);
        }
        
        // 初始化进度条
        updateProgress();
    </script>
